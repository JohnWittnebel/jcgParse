            this.list.forEach(function (item) {
                if (self.filter === 'all') {
                    self.listFiltered.push(item);
                } else if (self.filter === 'winner' && item.result) {
                    self.listFiltered.push(item);
                } else if (self.filter === 'loser' && !item.result) {
                    self.listFiltered.push(item);
                }
            });
            this.next();
            setTimeout(function () {
                self.onScroll();
            })
        },
        next() {
            if (this.loading) {
                return;
            }
            this.loading = true;
            for (let i = 0; i < this.maxViews; i++) {
                const item = this.listFiltered.shift();
                if (item) {
                    this.listVisible.push(item);
                }
            }
            const self = this;
            setTimeout(function () {
                self.loading = false;
            })
        },
        getProfileLink: function (nicename) {
            return 'https://sv.j-cg.com/user' + '/' + nicename;
        },
        getClassName: function (result) {
            if (this.bracketPublished) {
                return (result) ? 'entry winner' : 'entry loser';
            }
            return 'entry';
        },
        getClanImage: function (clanId) {
            return this.clanImages[clanId];
        },
        getDeckUrl: function (clanId, hash) {
	    if( clanId > 10 ){
                return 'https://shadowverse-portal.com/deck_co/' + hash;
	    }else{
                return 'https://shadowverse-portal.com/deck/' + hash;
	    }
        },
    }
</script>
